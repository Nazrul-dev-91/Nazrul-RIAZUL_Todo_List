# --- Django Project Start Guide ---

# 1. Create Virtual Environment (Only once at the start)
# python -m venv env

# 2. Activate Virtual Environment (Every time you start working)
# Windows: env\Scripts\activate
# Mac/Linux: source env/bin/activate

# 3. Install Required Libraries (If requirements.txt exists)
# pip install -r requirements.txt

# 4. Start New Project (If starting for the first time)
# django-admin startproject myproject . (The dot at the end creates files in the current folder)

# 5. Create New App (For every new feature)
# python manage.py startapp myapp

# 6. Database Migrations (After creating or changing models)
# python manage.py makemigrations
# python manage.py migrate

# 7. Create Superuser (For the admin panel)
# python manage.py createsuperuser

# 8. Run Server (While working on the project)
# python manage.py runserver

# 9. Update requirements.txt (After installing new libraries)
# pip freeze > requirements.txt
